CREATE TABLE User (idUser INT NOT NULL AUTO_INCREMENT, nom VARCHAR(255), prenom VARCHAR(255), login VARCHAR(255), password VARCHAR(255), PRIMARY KEY (idUser));
ALTER TABLE User ADD CONSTRAINT UQ_login UNIQUE(login);

CREATE TABLE UserList (idSurList INT NOT NULL AUTO_INCREMENT, titre VARCHAR(255), des VARCHAR(255), idUser INT NOT NULL, PRIMARY KEY (idSurList), dateCrea VARCHAR(255), dateModif VARCHAR(255));
ALTER TABLE UserList ADD CONSTRAINT fk_User FOREIGN KEY(idUser) REFERENCES User(idUser);

CREATE TABLE Element (idElement INT NOT NULL AUTO_INCREMENT, titre VARCHAR(255), des VARCHAR(255), idSurList INT, PRIMARY KEY (idElement), dateCrea VARCHAR(255), dateModif VARCHAR(255), etat VARCHAR(255));
ALTER TABLE Element ADD CONSTRAINT fk_List FOREIGN KEY(idSurList) REFERENCES UserList(idSurList);
